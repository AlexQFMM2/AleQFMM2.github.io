# 顺序表的合并

#### 1.头文件与符号常量的调用

```
#include<stdio.h>
#include <stdlib.h>
#define LIST_INIT_SIZE 100  
#define LISTINCREMENT 10
```

#### 2.顺序表的结构体定义

```
typedef struct SqList{   	//定义顺序表结构体,SqList为结构体标签 
	int *elem;		//顺序表的第一位 
	int length;		//顺序表的长度 
	int listsize;	//顺序表的字节 
}Sqlist; //顺序表的变量 
```

<!--more-->

#### 3.初始化顺序表

```
void initSqList(Sqlist *list){ //初始化顺序表 
	//创建一个足够大的整数数组，使顺序表结构体中的 elem 成员指向这个数组 
	list->elem = (int *)malloc(sizeof(int) * LIST_INIT_SIZE); 
	list->length = 0; //初始化长度 
	list->listsize = LIST_INIT_SIZE;	//初始化字节 
}
```

#### 4.顺序表输入

```
int data_scanf(SqList *list){
	printf("请输入不超过100个整数,输入任意其他字符结束：\n"); //从键盘输入值到顺序表 
	 int num;
	 do {
        if (scanf("%d",&num)!=1) break ;
        list->elem[list->length] = num;
    	list->length++;
    }while(list->length<LIST_INIT_SIZE);
    

//清楚输入缓存区 
int c;	 
while ((c = getchar()) != '\n' && c != EOF);

}
```

#### 5.顺序表输出

```
void data_print(Sqlist *list){ //顺序表输出 
	printf("\n顺序表目前的元素为:\n");
	for (int i = 0; i < list->length; i++) {
        printf("%d ", list->elem[i]);
    }
    printf("顺序表目前的长度为:%d\n",list->length);
}
```

#### 6.顺序表合并

```
void list_merge(Sqlist *L1,Sqlist *L2,Sqlist *L3){
	int *p1=L1->elem,*p2=L2->elem,*p3=L3->elem; //各做一个指针指向3个顺序表的开头 
	int *p1_last,*p2_last;
	L3->listsize=L3->length=L1->length+L2->length; //为L3长度字节重新赋值；
	p1_last = L1->elem + L1->length ; //定义L1，L2的末地址 
	p2_last = L2->elem + L2->length ;
	while(p1<p1_last&&p2<p2_last){ //逐一比较，小的放入L3 
		if(*p1 < *p2) *p3++ = *p1++;
		else *p3++ = *p2++;
	}
	while(p1<p1_last) *p3++ = *p1++;
	while(p2<p2_last) *p3++ = *p2++;
}
```

#### 7.main函数

```
int main()
{
	Sqlist Sq1,Sq2,Sq3;
	initSqList(&Sq1);
	initSqList(&Sq2);
	initSqList(&Sq3);
	data_scanf(&Sq1);
	data_scanf(&Sq2);
	data_print(&Sq1);
	data_print(&Sq2);
	list_merge(&Sq1,&Sq2,&Sq3);
	data_print(&Sq3);
}
```
