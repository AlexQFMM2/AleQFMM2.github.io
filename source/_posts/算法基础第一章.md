void QuickSort(vector<int>& nums,int l,int r) { //快速排序
			if(r-l+1 <=5) {	
				insertSort(nums,l,r);//用插入排序加个速
				return ;
			}
			int idx = l + rand()%(r-l+1);
			int piv = nums[idx];
			int left =l;
			int right =r;
			while(left <= right) {
				while(nums[left] < piv) ++left;
				while(nums[right] > piv) --right;
				if(left <= right) {
					swap(nums[left],nums[right]);
					++left;
					--right;
				}
			}
			QuickSort(nums,l,right);
			QuickSort(nums,left,r);
		}

# 二、快速选择

```
寻找指定序列中第k大的元素
```



```
class Solution {
public:
    int quickselect(vector<int> &nums, int l, int r, int k) {
        if (l == r)//左边界等于右边界找到了
            return nums[k];
        
        int flag = nums[ l+r >> 1], i = l - 1, j = r + 1;
        
        while (i < j) {//令flag左边全是小于他的，右边全是大于他的
            while (nums[++i] < flag);
            while (nums[--j] > flag);
            if (i < j)
                swap(nums[i], nums[j]);
        }
        
        //j :左边部分的末尾索引
        if (k <= j)return quickselect(nums, l, j, k);
        else return quickselect(nums, j + 1, r, k);
    }

    int findKthLargest(vector<int> &nums, int k) {
        int n = nums.size();
        return quickselect(nums, 0, n - 1, n - k);
    }
};

			if(r-l+1 <= 5){
				insertSort(nums,l,r);//利用插入提速
				return;
			} 
			
			int m = l + (r - l >> 1);
			mergeSort(nums, l, m); // 对左子数组进行排序
			mergeSort(nums, m + 1, r); // 对右子数组进行排序
			
			vector<int>temp(r-l+1);
			int a1=l,a2=m + 1,k=0;
			while(a1 <= m && a2 <=r){
				if(nums[a1] < nums[a2]) temp[k++] = nums[a1++];
				else temp[k++] = nums[a2++];
			}
			while(a1<=m)  temp[k++] = nums[a1++];
			while(a2<=r)  temp[k++] = nums[a2++];
			
			for(int i=0;i<k;i++)
				nums[l+i] = temp[i];
				
			return;
		}

# 四、逆序对的数量

```
逆序对 ： 一个集合中 符合 i>j && a[i] <a[j] 条件的组合 
```



```
#include <iostream>
using namespace std;
typedef long long LL;
const int N = 100010;
int n;
int a[N],tmp[N];
LL merge_sort (int a[],int l,int r) {  
    if (l == r) return 0;
    int mid = l + r >> 1;

    LL ans = merge_sort (a,l,mid) + merge_sort (a,mid + 1,r);

    int i = l,j = mid + 1,k = 0;
    while (i <= mid && j <= r) {
        if (a[i] <= a[j]) tmp[k++] = a[i++];
        else tmp[k++] = a[j++],ans += mid - i + 1;
    }

    while (i <= mid) tmp[k++] = a[i++];
    while (j <= r) tmp[k++] = a[j++];

    for (int i = 0;i<k;i++) a[i+l] = tmp[i];
    return ans;
}

int main () {
    cin >> n;
    for (int i = 0;i < n;i++) cin >> a[i];
    cout << merge_sort (a,0,n-1) << endl;
    return 0;
}

```



```
#include<iostream>
#include<vector>
using namespace std;

int search(vector<int>& nums,int target,int flag) { // 0代表起点，1代表终点
    int left =0,right = nums.size()-1;

    while(left <= right) {
        int mid = (left+right) >> 1;
        if(nums[mid] < target) left=mid+1;
        else if(nums[mid] > target) right=mid-1;
        else {
            if(flag) {
                if(nums[mid+1]==nums[mid]) {//找终点
                    left = mid + 1;
                    right = nums.size()-1;
                } else {
                    return mid;
                }
            } else {
                if(nums[mid-1]==nums[mid] ) {//找起点
                    left = 0;
                    right = mid - 1;
                } else {
                    return mid;
                }
            }
        }
    }
    return -1;
}

int main() {
    int n,k;
    cin >> n >> k;
    vector<int>nums(n);
    for(int i=0;i<n;i++)
        cin>>nums[i];
    while(k--){
        int target;
        cin >> target;
        int start = search(nums,target,0);
        int end = search(nums,target,1);
        cout << start << " " << end <<endl;
    }

    return 0;

# 九、差分

```
nums(n+1),sum(n+2);[1,n]

构造：sub[i] = nums[i] - nums[i-1]

对 [a,b]区间修改 c

sub[a] +=c;
sub[b+1] -=c;

还原：nums[i] = nums[i-1] + sub[i];
```



```
#include<iostream>

    }
    while(k--){
        int a,b,c;

        sub[b+1] -=c;

# 十、差分矩阵

```
vector<vector<int>> nums(row + 1, vector<int>(col + 1, 0));
vector<vector<int>> diff(row + 2, vector<int>(col + 2, 0));

构造 diff[i][j] = nums[i][j] - nums[i - 1][j] - nums[i][j - 1] + nums[i - 1][j - 1]; 

修改
diff[x1][y1] += c;
diff[x2 + 1][y1] -= c;
diff[x1][y2 + 1] -= c;
diff[x2 + 1][y2 + 1] += c;

还原：
nums[i][j] = diff[i][j] + nums[i - 1][j] + nums[i][j - 1] - nums[i - 1][j - 1];
```



```
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, m, k;
    cin >> n >> m >> k;
    
    vector<vector<int>> nums(n + 1, vector<int>(m + 1, 0));
    vector<vector<int>> diff(n + 2, vector<int>(m + 2, 0));
    
    // 读入原始矩阵
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cin >> nums[i][j];
        }
    }
    
    // 构建差分矩阵
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            diff[i][j] = nums[i][j] - nums[i - 1][j] - nums[i][j - 1] + nums[i - 1][j - 1];
        }
    }
    
    // 区间修改
    while (k--) {
        int x1, y1, x2, y2, c;
        cin >> x1 >> y1 >> x2 >> y2 >> c;
        
        diff[x1][y1] += c;
        diff[x2 + 1][y1] -= c;
        diff[x1][y2 + 1] -= c;
        diff[x2 + 1][y2 + 1] += c;
    }
    
    // 构建最终的矩阵
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            nums[i][j] = diff[i][j] + nums[i - 1][j] + nums[i][j - 1] - nums[i - 1][j - 1];
        }
    }
    
    // 输出最终矩阵
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cout << nums[i][j] << " ";
        }
        cout << endl;
    }
    
    return 0;
}

# 十一、最长连续不重复子序列

```
#include<iostream>
#include<vector>
#include<unordered_map>
#include<math.h> 
using namespace std;

int main(){
    int n;
    cin >> n;
    vector<int> nums(n+1,-1);
    unordered_map<int,int>idx;//利用hashmap存储出现过的元素的下标
    int ans = 0;

    for(int i=1,l=1;i<=n;i++){
        cin >> nums[i];
        if(idx.count(nums[i]) > 0){//出现重复元素
            int len = idx.size();//也可以用r-l
            ans = max(ans,len);
            while(l <= (idx[nums[i]])){//删除重复元素第一次出现前的所有元素
                idx.erase(nums[l++]);
            }
        }
        idx[nums[i]] = i;
    }

    int len = idx.size();
    ans = max(ans,len);//最后比一下末尾
    cout << ans;
}

# 十五、区间和

数据范围比较大，用离散映射存储，最后再差分

```
#include<iostream>
#include<vector>
#include <algorithm>
using namespace std;
using ll = long long;
using PILL = pair<ll,ll>;
const int N = 1e6;

ll find(vector<ll>&key,ll x){//寻找key中第一个大于等于x的元素
	ll l=0,r=key.size()-1;
	while(l<r){
		ll mid = l+r >> 1;
		if(key[mid] >= x) r=mid;
		else l = mid+1;
	}
	return r+1;
}

int main(){
	ll n,k;
	cin >> n >> k;
	vector<PILL>hash;	//key-value组合 
	vector<ll>key;		//关键字 
	vector<ll>value(N,0); //对应的值 
	
	for(ll i=1;i<=n;++i){
		ll x,c;
		cin >> x >> c;
		hash.push_back({x,c});
		key.push_back(x);
	}
	
	vector<PILL>test;//测试数据 
	for(ll i=1;i<=k;i++){
		ll l,r;
		cin >> l >> r;
		test.push_back({l,r});
		
		key.push_back(l);
		key.push_back(r);//(空key) 
	}
	
	sort(key.begin(),key.end());//对key进行一次排序 
	key.erase(unique(key.begin(),key.end()),key.end());//删除重复的key 
	
	for(auto it:hash){//记录value值
		value[find(key,it.first)] += it.second;
	}

		cout << valueSum[r] - valueSum[l-1] << endl;
	}
	
	return 0;
}
